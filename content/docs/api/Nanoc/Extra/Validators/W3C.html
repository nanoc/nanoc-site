def run
  args = []
  types = @types.dup
  args << 'html' if types.delete(:html)
  args << 'css'  if types.delete(:css)
  unless types.empty?
    raise Nanoc::Errors::GenericTrivial, "unknown type(s) specified: #{types.join(', ')}"
  end

  Nanoc::CLI.run([ 'check', args ].flatten)
end
