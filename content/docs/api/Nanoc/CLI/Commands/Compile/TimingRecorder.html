def start
  Nanoc::NotificationCenter.on(:filtering_started) do |rep, filter_name|
    @times[filter_name] ||= []
    @times[filter_name] << { :start => Time.now }
  end
  Nanoc::NotificationCenter.on(:filtering_ended) do |rep, filter_name|
    @times[filter_name].last[:stop] = Time.now
  end
end
