def self.from(obj)
  case obj
  when Nanoc::StringPattern, Nanoc::RegexpPattern
    obj
  when String
    Nanoc::StringPattern.new(obj)
  when Regexp
    Nanoc::RegexpPattern.new(obj)
  else
    raise ArgumentError, "Do not know how to convert #{obj} into a Nanoc::Pattern"
  end
end
