def [](s)
  case s
  when Nanoc::Identifier
    item_with_identifier(s)
  when String
    item_with_identifier(s) || glob_single(s)
  else
    raise ArgumentError, 'Can only call ItemCollection#[] with string or identifier'
  end
end

def fetch(patternish)
  res = self[patternish]
  if res.nil?
    raise Nanoc::Errors::NoObjectsMatchingPattern.new('item', patternish)
  else
    res
  end
end
