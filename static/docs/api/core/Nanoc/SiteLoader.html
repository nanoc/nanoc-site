def load
  # Load
  config
  code_snippets
  data_sources
  data_sources.each { |ds| ds.use }
  items
  layouts
  data_sources.each { |ds| ds.unuse }

  # Build site
  Nanoc::Site.new({
    data_sources:  data_sources,
    items:         items,
    layouts:       layouts,
    code_snippets: code_snippets,
    config:        config,
  })
end
