Overview

Used for identifying items and layouts.

An identifier resembles a filesystem path quite closely: it is a list of strings separated by slashes. The last component can have an extension.

Some examples of identifiers:

/doc/tutorial.md
/index.html

def self.coerce(input)
  if input.is_a?(String)
    from_string(input)
  elsif input.is_a?(Nanoc::Identifier)
    return input
  else
    raise Nanoc::Errors::Generic "Do not know how to coerce #{input.inspect} into a Nanoc::Identifier"
  end
end

def self.from_string(string)
  if string =~ /\/$/
    raise Nanoc::Errors::IdentifierCannotEndWithSlashError.new(string)
  end

  components = string.split('/').reject { |c| c.empty? }
  new(components)
end

def extension
  c = components[-1]
  idx = c.rindex('.')
  if idx
    c[idx + 1..-1]
  else
    nil
  end
end

def parent

  if parent_components.empty?

  else
    self.class.new(parent_components)

def with_ext(ext)
  cs = without_ext.components.dup
  cs[-1] = cs[-1] + '.' + ext
  self.class.new(cs)
end

def without_ext
