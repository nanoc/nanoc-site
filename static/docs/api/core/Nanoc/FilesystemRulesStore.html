def load_rules
  # Get rule data
  if !File.file?(rules_filename)
    raise Nanoc::Errors::NoRulesFileFound.new
  end
  @rule_data = File.read(rules_filename)

  # Load DSL
  dsl = Nanoc::CompilerDSL.new(rules_collection)
  dsl.instance_eval(@rule_data, "./#{rules_filename}")
end
